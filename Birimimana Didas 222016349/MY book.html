        • SQL QUERIES

        1. Applicant_Info

          2. CREATE TABLE Applicant_Info (
          Student_ID INT PRIMARY KEY,
          Father_Name VARCHAR(50),
          Mother_Name VARCHAR(50),
          Email VARCHAR(100),
          Phone_Number VARCHAR(20),
          Address VARCHAR(100),
          Age INT
          );
        2. Course_Info

          3. CREATE TABLE Course_Info (
          Student_ID INT,
          Course_Title VARCHAR(100),
          Department VARCHAR(50),
          Number_of_Credits INT,
          Semester_Offered VARCHAR(50),
          Grade VARCHAR(2),
          FOREIGN KEY (Student_ID) REFERENCES Applicant_Info(Student_ID)
          );
        3. Course_Info

          4. CREATE TABLE Project Research Info(
          CREATE TABLE Project_Research_Info (
          Project_Name VARCHAR(100) PRIMARY KEY,
          Project_Purpose VARCHAR(500),
          Student_ID INT,
          Project_Invigilator_Name VARCHAR(100),
          FOREIGN KEY (Student_ID) REFERENCES Applicant_Info(Student_ID)

        4. SQL codes for Semester Info


        CREATE TABLE Semester_Info (
        Semester_Info_ID INT PRIMARY KEY,
        Start_Date DATE,
        Ending_Date DATE,
        Student_ID INT,
        Semester_Code VARCHAR(10),
        Total_Number_of_Credits INT,
        FOREIGN KEY (Student_ID) REFERENCES Applicant_Info(Student_ID)
        );
